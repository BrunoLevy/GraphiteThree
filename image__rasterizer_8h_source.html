92  void set_pixel(int x, int y, const Color& c) {
93  geo_debug_assert(x >= 0 && x < int(image_->width()));
94  geo_debug_assert(y >= 0 && y < int(image_->height()));
95  switch(component_encoding_) {
96  case Image::BYTE: {
97  Memory::byte* pixel_ptr =
98  (Memory::byte*)(
99  image_->pixel_base(index_t(x),index_t(y))
100  );
101  for(index_t comp=0; comp<nb_components_; ++comp) {
102  pixel_ptr[comp] = Memory::byte(c[comp] * 255.0);
103  }
104  } break;
105  case Image::FLOAT32: {
106  Numeric::float32* pixel_ptr =
107  (Numeric::float32*)(void*)(
108  image_->pixel_base(index_t(x),index_t(y))
109  );
110  for(index_t comp=0; comp<nb_components_; ++comp) {
111  pixel_ptr[comp] = Numeric::float32(c[comp]);
112  }
113  } break;
114  case Image::FLOAT64: {
115  Numeric::float64* pixel_ptr =
116  (Numeric::float64*)(void*)(

118  );

121  }

124  case Image::INT32: {

153  void interpolate_color(
154  const Color& c1, const Color& c2, const Color& c3,
155  double l1, double l2, double l3,
156  Color& c
157  ) {
158  c[0] = l1*c1[0] + l2*c2[0] + l3*c3[0];
159  c[1] = l1*c1[1] + l2*c2[1] + l3*c3[1];
160  c[2] = l1*c1[2] + l2*c2[2] + l3*c3[2];
161  c[3] = l1*c1[3] + l2*c2[3] + l3*c3[3];
162  }
