192  bool halfedge_is_valid(const Halfedge& H) const {
193  return
194  H.facet != Halfedge::NO_FACET &&
195  H.corner != Halfedge::NO_CORNER &&
196  H.facet < mesh_.facets.nb() &&
197  H.corner < mesh_.facet_corners.nb()

209  bool halfedge_is_border(const Halfedge& H) const {
210  geo_debug_assert(halfedge_is_valid(H));
211  if(facet_region_.is_bound()) {
212  index_t f = H.facet;
213  index_t adj_f =
214  mesh_.facet_corners.adjacent_facet(H.corner);
215  return
216  adj_f == NO_FACET ||
217  facet_region_[f] != facet_region_[adj_f]
218  ;
219  }
220  return mesh_.facet_corners.adjacent_facet(H.corner) == NO_FACET;
221  }
222
