152  template <typename VECTOR> inline void sort_unique(VECTOR& v) {
153  std::sort(v.begin(), v.end());
154  // Note that std::unique leaves a 'queue' of duplicated elements
155  // at the end of the vector, and returns an iterator that
156  // indicates where to stop.
157  v.erase(
158  std::unique(v.begin(), v.end()), v.end()

160  }
