129  void new_facet(
130  index_t tglobal, index_t boundary_f,
131  signed_index_t v0, signed_index_t v1, signed_index_t v2
132  ) {
133  if(!OK_) {
134  return;
135  }
136 
137  geo_debug_assert(v0 != v1);
138  geo_debug_assert(v1 != v2);
139  geo_debug_assert(v2 != v0);
140 
141  local_index_t new_t = local_index_t(nb_f_);
142 
143  if(nb_f_ == MAX_F) {
144  OK_ = false;
145  return;
146  }
147 
148  set_vv2t(v0, v1, new_t);
149  set_vv2t(v1, v2, new_t);
150  set_vv2t(v2, v0, new_t);
151 
152  if(!OK_) {
153  return;
154  }
155 
156  ++nb_f_;
157  tglobal_[new_t] = tglobal;
158  boundary_f_[new_t] = boundary_f;
159  f2v_[new_t][0] = v0;
160  f2v_[new_t][1] = v1;
161  f2v_[new_t][2] = v2;
162  }

244  index_t hash(signed_index_t v1, signed_index_t v2) const {
245  return ((index_t(v1+1) ^ (419*index_t(v2+1))) % MAX_H);

254  void set_vv2t(
255  signed_index_t v1, signed_index_t v2, local_index_t f
256  ) {
257  CAVITY_STATS(index_t cnt = 0;)
258  index_t h = hash(v1,v2);
259  index_t cur = h;
260  do {
261  if(h2t_[cur] == END_OF_LIST) {
262  h2t_[cur] = f;
263 #ifdef GARGANTUA
264  h2v_[cur][0] = v1;
265  h2v_[cur][1] = v2;
266 #else
267  h2v_[cur] = (Numeric::uint64(v1+1) << 32) | Numeric::uint64(v2+1);
268 #endif
269  CAVITY_STATS(++stats_set_[cnt];)
270  return;
271  }
272  cur = (cur+1)%MAX_H;
273  CAVITY_STATS(++cnt;)
274  } while(cur != h);
275  OK_ = false;
276  }

284  local_index_t get_vv2t(signed_index_t v1, signed_index_t v2) const {
285 #ifndef GARGANTUA
286  Numeric::uint64 K = (Numeric::uint64(v1+1) << 32) | Numeric::uint64(v2+1);
287 #endif
288  CAVITY_STATS(index_t cnt = 0;)
289  index_t h = hash(v1,v2);
290  index_t cur = h;
291  do {
292 #ifdef GARGANTUA
293  if((h2v_[cur][0] == v1) && (h2v_[cur][1] == v2)) {
294 #else
295  if(h2v_[cur] == K) {
296 #endif
297  CAVITY_STATS(++stats_get_[cnt];)
298  return h2t_[cur];
299  }
300  cur = (cur+1)%MAX_H;
301  CAVITY_STATS(++cnt;)
302  } while(cur != h);
303  geo_assert_not_reached;
304  }
