108  template <class T> inline bool read_ascii_attribute(
109  FILE* file, Memory::pointer base_addr, index_t nb_elements
110  ) {
111  T* attrib = reinterpret_cast<T*>(base_addr);
112  for(index_t i=0; i<nb_elements; ++i) {
113  std::string buff;
114  int res;
115  while(char(res = fgetc(file)) != '\n') {
116  if(res == EOF) {
117  return false;
118  }
119  buff.push_back(char(res));
120  }
121  if(!String::from_string(buff.c_str(),attrib[i])) {
122  return false;
123  }
124  }
125  return true;
126  }
