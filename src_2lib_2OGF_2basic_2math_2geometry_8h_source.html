74  template <class FT> vecng<2,FT> transform_point(
75  const vecng<2,FT>& v,
76  const Matrix<4,FT>& m
77  ) {
78  FT result[4] ;
79  FT w[4] ;
80  index_t i,j ;
81 
82  for(i=0; i<4; i++) {
83  result[i] = 0 ;
84  }
85  w[0] = v.x ;
86  w[1] = v.y ;
87  w[2] = FT(0) ;
88  w[3] = FT(1) ;
89 
90  for(i=0; i<4; i++) {
91  for(j=0; j<4; j++) {
92  result[i] += w[j] * m(j,i) ;
93  }
94  }
95 
96  return vecng<2,FT> (
97  result[0] / result[3],
98  result[1] / result[3]
99  ) ;
100  }
